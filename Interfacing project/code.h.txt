// ============================================================
// ISRs
// ============================================================
void IRAM_ATTR readLeftEncoder() {                 // Interrupt for left encoder
  if (digitalRead(ENC_LEFT_A) == digitalRead(ENC_LEFT_B)) leftEncoderTicks++; // If A == B, count up
  else leftEncoderTicks--;                          // Else count down (quadrature decoding)
}
void IRAM_ATTR readRightEncoder() {                // Interrupt for right encoder
  if (digitalRead(ENC_RIGHT_A) == digitalRead(ENC_RIGHT_B)) rightEncoderTicks++; // Same logic
  else rightEncoderTicks--;
}

// ============================================================
// Motor control
// ============================================================
void setMotors(int L, int R) {                     // Set left and right motor speeds (-255 to 255)
  L = constrain(L, -255, 255);                     // Constrain left speed to valid range
  R = constrain(R, -255, 255);                     // Constrain right speed
  if      (L > 0) { analogWrite(IN1, L);   analogWrite(IN2, 0); } // Forward: IN1 PWM, IN2 low
  else if (L < 0) { analogWrite(IN1, 0);   analogWrite(IN2, -L); } // Reverse: IN1 low, IN2 PWM with positive value
  else            { analogWrite(IN1, 0);   analogWrite(IN2, 0); } // Stop: both low
  if      (R > 0) { analogWrite(IN3, R);   analogWrite(IN4, 0); } // Same for right motor
  else if (R < 0) { analogWrite(IN3, 0);   analogWrite(IN4, -R); }
  else            { analogWrite(IN3, 0);   analogWrite(IN4, 0); }
}

void stopMotors() {                                 // Emergency stop with braking
  setMotors(-brakeStrength, -brakeStrength);       // Apply reverse PWM to both motors
  delay(brakeDurationMs);                          // Hold brake for specified duration
  setMotors(0, 0);                                  // Finally stop
}

// ============================================================
// Encoder helpers
// ============================================================
float encoderAvgDistanceMM() {                      // Compute average distance traveled from encoders (mm)
  float dL = (abs((float)leftEncoderTicks)  / PPR) * wheelCircumference; // Left wheel distance
  float dR = (abs((float)rightEncoderTicks) / PPR) * wheelCircumference; // Right wheel distance
  return (dL + dR) / 2.0f;                           // Return average
}

void resetEncoders() {                               // Reset encoder counts to zero
  leftEncoderTicks  = 0;
  rightEncoderTicks = 0;
}

// ============================================================
// 3-SENSOR FUSION STRAIGHT-LINE CORRECTION
// ============================================================
int fusedStraightCorrectionFull() {                  // Compute motor correction using all sensors
  // Layer 1: Gyro
  float gyroErr   = currentYaw - headingLock;       // Error: deviation from locked heading
  float gyroDeriv = gyroErr - lastGyroErr;          // Derivative of error
  lastGyroErr     = gyroErr;                         // Store for next time
  float gyroCorr  = -(gyroKp * gyroErr + gyroKd * gyroDeriv); // PD control (negative sign for direction)
  gyroCorr = constrain(gyroCorr, -(float)gyroMaxCorr, (float)gyroMaxCorr); // Limit correction
  live_fc_gyro = gyroCorr;                            // Store for display

  // Layer 2: Encoder
  long relLeft  = leftEncoderTicks  - encLeftStart;  // Left encoder ticks since start of move
  long relRight = rightEncoderTicks - encRightStart; // Right encoder ticks since start
  long encErr   = relLeft - relRight;                 // Difference indicates turning
  float encCorr = encKp * (float)encErr;              // Proportional correction
  encCorr = constrain(encCorr, -(float)encMaxCorr, (float)encMaxCorr); // Limit
  live_fc_enc = encCorr;                               // Store

  // Layer 3: LiDAR (Absolute Single-Wall Tracking)
  float wallCorr = 0.0f;                               // Initialize wall correction
  bool hasLeft  = (distLeft  > 0 && distLeft  < wallSensorLimit); // Valid left wall?
  bool hasRight = (distRight > 0 && distRight < wallSensorLimit); // Valid right wall?
  bool wallsValid = (hasLeft || hasRight);             // At least one wall detected

  // GEOMETRY CALCULATION:
  // Node is 200mm wide. Robot is 100mm wide. 
  // (200 - 100) / 2 = 50mm distance to each wall.
  // Adjust this slightly if your sensors are mounted further inside the chassis.
  float idealWallDist = 50.0f;                          // Desired distance to a single wall
  int wallErr = 0;                                       // Wall error

  if (hasLeft && hasRight) {                             // Both walls present
    // Normal centering: difference should be zero if centered
    wallErr = distRight - distLeft;                      // Positive if robot too far left (right wall closer)
  } else if (hasLeft && !hasRight) {                     // Only left wall
    // Push towards/away from left wall to maintain ideal distance
    // Multiplied by 2 to match the error magnitude of the two-wall state
    wallErr = (idealWallDist - distLeft) * 2;            // Positive if too far from left wall
  } else if (!hasLeft && hasRight) {                     // Only right wall
    wallErr = (distRight - idealWallDist) * 2;           // Positive if too close to right wall
  }

  if (wallsValid) {                                       // If at least one wall detected
    if (abs(wallErr) > wallDeadband) {                    // Error exceeds deadband
      int wallDeriv = wallErr - lastWallErr;              // Derivative
      wallCorr = wallKp * wallErr + wallKd * (float)wallDeriv; // PD control
      wallCorr = constrain(wallCorr, -(float)wallMaxCorr, (float)wallMaxCorr); // Limit
    }
    lastWallErr = wallErr;                                 // Update for next derivative
  } else {
    lastWallErr = 0;                                       // No walls, reset error
  }
  live_fc_lidar = wallCorr;                                // Store

  // Weights
  float wG = wGyro, wE = wEncoder, wL = wLidar;           // Start with configured weights
  if (!wallsValid) {                                        // If no walls detected
    float r = wLidar;                                       // Amount to redistribute
    wG = wGyro    + r * (wGyro    / (wGyro + wEncoder));   // Redistribute LiDAR weight to gyro and encoder proportionally
    wE = wEncoder + r * (wEncoder / (wGyro + wEncoder));
    wL = 0.0f;                                              // No LiDAR weight
  }

  float fused = (wG * gyroCorr) + (wE * encCorr) + (wL * wallCorr); // Weighted sum
  fused = constrain(fused, -(float)fusedMaxCorr, (float)fusedMaxCorr); // Limit total correction
  live_fc_fused = fused;                                   // Store
  return (int)fused;                                        // Return as integer PWM adjustment
}

void driveFused(int spd) {                                  // Drive forward with fused correction
  int corr = fusedStraightCorrectionFull();                 // Get correction value
  setMotors(constrain(spd - corr, spd - fusedMaxCorr, spd + fusedMaxCorr), // Apply correction to left motor (subtract)
            constrain(spd + corr, spd - fusedMaxCorr, spd + fusedMaxCorr)); // Apply to right (add)
}

void driveBackwardFused(int spd) {                          // Drive backward with fused correction
  int corr = fusedStraightCorrectionFull();                 // Get correction (same as forward)
  setMotors(constrain(-spd - corr, -(spd + fusedMaxCorr), -(spd - fusedMaxCorr)), // For backward, base speed negative, correction subtracts from left
            constrain(-spd + corr, -(spd + fusedMaxCorr), -(spd - fusedMaxCorr))); // Correction adds to right
}

void lockHeading() {                                         // Reset all straight-line tracking variables
  headingLock   = currentYaw;                                // Set reference heading to current yaw
  lastGyroErr   = 0.0f;                                      // Reset gyro error
  lastWallErr   = 0;                                         // Reset wall error
  encLeftStart  = leftEncoderTicks;                          // Record starting encoder counts
  encRightStart = rightEncoderTicks;
  live_fc_gyro = live_fc_enc = live_fc_lidar = live_fc_fused = 0; // Reset displayed corrections
}

bool isTarget(int r, int c) {                                 // Check if cell (r,c) is in target area (center 2x2)
  return (r == 3 || r == 4) && (c == 3 || c == 4);           // True for rows 3,4 and cols 3,4
}

void setWallBilateral(int r, int c, int d) {                  // Set wall in given direction, also in neighboring cell opposite wall
  if (r < 0 || r >= 8 || c < 0 || c >= 8) return;             // Out of bounds, ignore
  mazeWalls[r][c] |= (1 << d);                                 // Set wall bit for this cell in direction d
  int nr = r + DR[d];                                          // Neighbor row
  int nc = c + DC[d];                                          // Neighbor column
  if (nr >= 0 && nr < 8 && nc >= 0 && nc < 8) {                // If neighbor exists
    mazeWalls[nr][nc] |= (1 << ((d + 2) % 4));                 // Set opposite wall in neighbor
  }
}

void initMaze() {                                              // Initialize maze arrays to default
  memset(mazeWalls, 0, sizeof(mazeWalls));                     // Clear all walls
  memset(mazeFlood, 255, sizeof(mazeFlood));                   // Set flood values to max (unreached)
  memset(mazeVisited, 0, sizeof(mazeVisited));                 // Mark all cells as unvisited
  pathLen = 0;                                                  // Reset path length
  // Outer boundary walls
  for (int i = 0; i < 8; i++) {                                 // For each cell on border
    mazeWalls[0][i] |= WALL_N;                                  // Top row has north wall
    mazeWalls[7][i] |= WALL_S;                                  // Bottom row has south wall
    mazeWalls[i][0] |= WALL_W;                                  // Left column has west wall
    mazeWalls[i][7] |= WALL_E;                                  // Right column has east wall
  }
}

void floodFill() {                                              // Perform flood fill from target area
  memset(mazeFlood, 255, sizeof(mazeFlood));                   // Reset flood values
  // BFS queue
  uint8_t qr[64], qc[64];                                       // Queue arrays (max 64 cells)
  int front = 0, back = 0;                                      // Queue pointers
  // Seed: center 4 target cells
  for (int r = 3; r <= 4; r++) {                                 // For rows 3 and 4
    for (int c = 3; c <= 4; c++) {                               // For cols 3 and 4
      mazeFlood[r][c] = 0;                                       // Set flood value to 0 at target
      qr[back] = r; qc[back] = c; back++;                        // Add to queue
    }
  }

  while (front < back) {                                         // While queue not empty
    int r = qr[front];                                           // Current row
    int c = qc[front]; front++;                                  // Current column, advance front
    uint8_t nextVal = mazeFlood[r][c] + 1;                        // Value for neighbors
    for (int d = 0; d < 4; d++) {                                 // Check all four directions
      if (mazeWalls[r][c] & (1 << d)) continue;                  // If wall in this direction, skip
      // wall blocks
      int nr = r + DR[d];                                         // Neighbor row
      int nc = c + DC[d];                                         // Neighbor column
      if (nr < 0 || nr >= 8 || nc < 0 || nc >= 8) continue;      // Out of bounds
      if (mazeFlood[nr][nc] > nextVal) {                          // If neighbor has larger flood value
        mazeFlood[nr][nc] = nextVal;                               // Update to smaller value
        qr[back] = nr;                                             // Add neighbor to queue
        qc[back] = nc; back++;
      }
    }
  }
}

void scanMazeWalls() {                                            // Read LiDARs to detect walls around current cell
  int frontDir = robotDir;                                        // Front direction is current robot direction
  int rightDir = (robotDir + 1) % 4;                              // Right direction
  int leftDir  = (robotDir + 3) % 4;                              // Left direction
  // Take multiple readings for reliability
  const int NUM_SAMPLES = 5;                                      // Number of samples
  int frontHits = 0, leftHits = 0, rightHits = 0;                 // Counters for each wall detection
  for (int s = 0; s < NUM_SAMPLES; s++) {                         // Loop for each sample
    // Read fresh sensor data
    if (lidarFront.isRangeComplete()) distFront = lidarFront.readRange(); // Update front distance
    if (lidarLeft.isRangeComplete())  distLeft  = lidarLeft.readRange();  // Update left distance
    if (lidarRight.isRangeComplete()) distRight = max(0, lidarRight.readRange() - rightSensorOffset); // Update right distance with offset
    if (distFront > 0 && distFront < wallDetectDist) frontHits++; // If front distance within wall threshold, count
    if (distLeft  > 0 && distLeft  < wallDetectDist) leftHits++;  // Left
    if (distRight > 0 && distRight < wallDetectDist) rightHits++; // Right

    delay(20);                                                      // Small delay between samples
    // small delay between samples
  }

  // Majority vote: only mark wall if 3+ out of 5 samples agree
  int threshold = (NUM_SAMPLES / 2) + 1; // 3

  if (frontHits >= threshold)                                      // If front wall detected in majority
    setWallBilateral(robotRow, robotCol, frontDir);                // Set front wall
  if (rightHits >= threshold)                                      // If right wall detected
    setWallBilateral(robotRow, robotCol, rightDir);                // Set right wall
  if (leftHits >= threshold)                                       // If left wall detected
    setWallBilateral(robotRow, robotCol, leftDir);                 // Set left wall

  addLog("MZ walls F:" + String(frontHits) + " L:" + String(leftHits) + // Log detection results
         " R:" + String(rightHits) + "/" + String(NUM_SAMPLES));
  mazeVisited[robotRow][robotCol] = true;                           // Mark current cell as visited
}

void addToPath() {                                                 // Add current position to path recording
  if (pathLen < 256) {                                              // If path buffer not full
    pathR[pathLen] = robotRow;                                      // Store row
    pathC[pathLen] = robotCol;                                      // Store column
    pathLen++;                                                       // Increment length
  }
}

// Find the best direction to move from current cell (lowest flood value neighbor)
// Priority on tie: straight > right > left > back
int findBestDirection() {                                           // Determine next direction based on flood fill
  int bestDir = -1;                                                 // Initialize with invalid
  int bestFlood = 255;                                              // Start with high flood value

  int checkOrder[4];                                                // Order of preference
  checkOrder[0] = robotDir;               // straight
  checkOrder[1] = (robotDir + 1) % 4;     // right
  checkOrder[2] = (robotDir + 3) % 4;     // left
  checkOrder[3] = (robotDir + 2) % 4;     // back

  for (int i = 0; i < 4; i++) {                                     // Check in priority order
    int d = checkOrder[i];                                           // Direction to check
    if (mazeWalls[robotRow][robotCol] & (1 << d)) continue;         // If wall, skip
    int nr = robotRow + DR[d];                                       // Neighbor row
    int nc = robotCol + DC[d];                                       // Neighbor column
    if (nr < 0 || nr >= 8 || nc < 0 || nc >= 8) continue;           // Out of bounds
    if (mazeFlood[nr][nc] < bestFlood) {                             // If flood value is lower than current best
      bestFlood = mazeFlood[nr][nc];                                 // Update best flood
      bestDir = d;                                                   // Update best direction
    }
  }
  return bestDir;                                                    // Return best direction (or -1 if none)
}

    case MAZE_TURN_PIVOT:          return "MZ_T_PIVOT";

    case MAZE_TURN_CENTER:         return "MZ_T_CTR";
    case MAZE_FORWARD:             return "MZ_FWD";
    case MAZE_REVERSE:             return "MZ_REV";
    case MAZE_DONE:                return "MZ_DONE";
    default:                       return "UNKNOWN";
  }
}

String mazeStateName() {                                             // Return simplified maze status
  if (!mazeActive && robotState != MAZE_DONE) return "IDLE";        // If not active and not done, idle
  switch (robotState) {                                              // Based on current state
    case MAZE_SCAN:          return "SCANNING";
    case MAZE_DECIDE:        return "DECIDING";
    case MAZE_TURN_BACKUP:   return "TURNING";
    case MAZE_TURN_PIVOT:    return "TURNING";
    case MAZE_TURN_ADVANCE:  return "TURNING";
    case MAZE_TURN_CENTER:   return "CENTERING";
    case MAZE_FORWARD:       return "MOVING";
    case MAZE_REVERSE:       return "REVERSING";
    case MAZE_DONE:          return "SOLVED";
    default:                 return mazeActive ? "ACTIVE" : "IDLE"; // Default if in other states but maze active
  }
}

// ============================================================

// ============================================================
void setup() {
  Serial.begin(115200);                     // Initialize serial communication for debugging

  pinMode(ENC_LEFT_A,  INPUT); pinMode(ENC_LEFT_B,  INPUT);  // Set encoder pins as inputs
  pinMode(ENC_RIGHT_A, INPUT); pinMode(ENC_RIGHT_B, INPUT);
  attachInterrupt(digitalPinToInterrupt(ENC_LEFT_A),  readLeftEncoder,  CHANGE);  // Attach interrupt for left encoder on CHANGE
  attachInterrupt(digitalPinToInterrupt(ENC_RIGHT_A), readRightEncoder, CHANGE);  // Attach interrupt for right encoder

  pinMode(IN1, OUTPUT); pinMode(IN2, OUTPUT);                // Set motor driver pins as outputs
  pinMode(IN3, OUTPUT); pinMode(IN4, OUTPUT);
  setMotors(0, 0);                                            // Ensure motors are stopped initially
  pinMode(XSHUT_LEFT,  OUTPUT);                               // Set XSHUT pins as outputs
  pinMode(XSHUT_FRONT, OUTPUT);
  pinMode(XSHUT_RIGHT, OUTPUT);
  digitalWrite(XSHUT_LEFT,  LOW);                             // Disable all LiDARs initially (LOW = shutdown)
  digitalWrite(XSHUT_FRONT, LOW);
  digitalWrite(XSHUT_RIGHT, LOW);
  delay(10);                                                   // Short delay to settle

  Wire.begin(GYRO_SDA,  GYRO_SCL);                             // Initialize I2C for MPU6050 (gyro)
  Wire1.begin(LIDAR_SDA, LIDAR_SCL);                           // Initialize second I2C for LiDARs
  if (!mpu.begin(0x68, &Wire)) {                               // Attempt to initialize MPU6050
    addLog("ERR: MPU6050");                                    // Log error if fails
  } else {
    mpu.setGyroRange(MPU6050_RANGE_500_DEG);                   // Set gyro range to ±500 deg/s
    mpu.setFilterBandwidth(MPU6050_BAND_21_HZ);                // Set low-pass filter to 21 Hz
    addLog("MPU6050 OK");                                       // Log success
  }

  digitalWrite(XSHUT_LEFT,  HIGH); delay(10);                  // Enable left LiDAR (HIGH = out of shutdown)
  if (!lidarLeft.begin(0x30,  false, &Wire1)) addLog("ERR: LiDAR L"); else addLog("LiDAR L OK"); // Initialize left LiDAR at address 0x30
  digitalWrite(XSHUT_FRONT, HIGH); delay(10);                  // Enable front LiDAR
  if (!lidarFront.begin(0x31, false, &Wire1)) addLog("ERR: LiDAR F"); else addLog("LiDAR F OK"); // Initialize front at 0x31
  digitalWrite(XSHUT_RIGHT, HIGH); delay(10);                  // Enable right LiDAR
  if (!lidarRight.begin(0x29, false, &Wire1)) addLog("ERR: LiDAR R"); else addLog("LiDAR R OK"); // Initialize right at 0x29

  lidarLeft.startRangeContinuous();                             // Start continuous ranging for all LiDARs
  lidarFront.startRangeContinuous();
  lidarRight.startRangeContinuous();

  WiFi.begin(ssid, password);                                   // Connect to WiFi
  int att = 0;                                                   // Attempt counter
  while (WiFi.status() != WL_CONNECTED && att < 20) {           // Wait up to 20 attempts (10 seconds)
    delay(500); Serial.print("."); att++;                       // Print dot each 500ms
  }
  if (WiFi.status() == WL_CONNECTED) {                          // If connected
    addLog("WiFi: " + WiFi.localIP().toString());               // Log IP address
    Serial.println("\nIP: " + WiFi.localIP().toString());
  } else {
    addLog("WiFi FAILED");                                       // Log failure
  }

  // Initialize maze
  initMaze();                                                    // Set up maze walls and flood
  floodFill();                                                   // Initial flood fill

  // ---- HTTP ROUTES ----
  server.on("/", []() { sendHtmlPage(); });                      // Serve main HTML page

  // ---- /data — combined sensor + maze JSON ----
  server.on("/data", []() {
    // Build wall array
    String wj = "["; String fj = "["; String vj = "[";
    for (int i = 0; i < 64; i++) {                               // Loop over all 64 cells
      if (i > 0) { wj += ","; fj += ","; vj += ","; }           // Add commas between elements
      wj += String(mazeWalls[i/8][i%8]);                         // Add wall value for this cell
      fj += String(mazeFlood[i/8][i%8]);                          // Add flood value
      vj += String(mazeVisited[i/8][i%8] ? 1 : 0);               // Add visited flag (1/0)
    }
    wj += "]"; fj += "]"; vj += "]";                              // Close arrays

    // Path array
    String pj = "[";
    for (int i = 0; i < pathLen; i++) {                           // For each step in path
      if (i > 0) pj += ",";
      pj += "[" + String(pathR[i]) + "," + String(pathC[i]) + "]"; // Add [row,col] pair
    }
    pj += "]";

    // Log
    String lj = "[";
    int count = min(logIndex, LOG_SIZE);                          // Number of log entries available
    
    int start = (logIndex >= LOG_SIZE) ? (logIndex % LOG_SIZE) : 0; // Determine start index in circular buffer
    for (int i = 0; i < count; i++) {
      int idx = (start + i) % LOG_SIZE;                           // Get actual index
      lj += "\"" + logBuffer[idx] + "\"";                         // Add log string in quotes
      if (i < count - 1) lj += ",";                               // Add comma if not last
    }
    lj += "]";

    String json =
      "{\"front\":"     + String(distFront)        +
      ",\"left\":"      + String(distLeft)          +
      ",\"right\":"     + String(distRight)         +
      ",\"yaw\":"       + String(currentYaw)        +
      ",\"hlock\":"     + String(headingLock)       +
      ",\"gz\":"        + String(gyroZ_raw)         +
      ",\"enc_left\":"  + String(leftEncoderTicks)  +
      ",\"enc_right\":" + String(rightEncoderTicks) +
      ",\"tgt_dist\":"  + String(targetDistanceMM)  +
      ",\"tgt_angle\":" + String(targetAngleDeg)    +
      ",\"fc_gyro\":"   + String(live_fc_gyro)      +
      ",\"fc_enc\":"    + String(live_fc_enc)       +
      ",\"fc_lidar\":"  + String(live_fc_lidar)     +
      ",\"fc_fused\":"  + String(live_fc_fused)     +
      ",\"state\":\""   + stateName()               + "\""  +
      ",\"ms2\":\""     + mazeStateName()           + "\""  +
      ",\"mr\":"        + String(robotRow)          +
      ",\"mc\":"        + String(robotCol)          +
      ",\"md\":"        + String(robotDir)          +
      ",\"ma\":"        + String(mazeActive ? 1 : 0) +
      ",\"mw\":"        + wj +
      ",\"mf\":"        + fj +
      ",\"mv\":"        + vj +
      ",\"mp\":"        + pj +
      ",\"log\":"       + lj + "}";
    server.send(200, "application/json", json);                   // Send JSON response
  });

  // ---- /settings ----
  server.on("/settings", []() {
    if (server.hasArg("ms")) moveSpeed = server.arg("ms").toInt(); // Update move speed if provided
    if (server.hasArg("ts")) turnSpeed = server.arg("ts").toInt(); // Update turn speed
    addLog("Speed mv=" + String(moveSpeed) + " trn=" + String(turnSpeed)); // Log change
    server.send(200, "text/plain", "OK");                         // Acknowledge
  });

  // ---- /fusioncfg ----
  server.on("/fusioncfg", []() {
    if (server.hasArg("wg"))  wGyro        = server.arg("wg").toFloat();   // Update gyro weight
    if (server.hasArg("we"))  wEncoder     = server.arg("we").toFloat();   // Update encoder weight
    if (server.hasArg("wl"))  wLidar       = server.arg("wl").toFloat();   // Update LiDAR weight
    if (server.hasArg("fmc")) fusedMaxCorr = server.arg("fmc").toInt();    // Update max fused correction

    server.send(200, "text/plain", "OK");
  });

  // ---- /correctcfg ----
  server.on("/correctcfg", []() {
    if (server.hasArg("gkp")) gyroKp       = server.arg("gkp").toFloat(); // Gyro Kp
    if (server.hasArg("gkd")) gyroKd       = server.arg("gkd").toFloat(); // Gyro Kd
    if (server.hasArg("gmc")) gyroMaxCorr  = server.arg("gmc").toInt();   // Gyro max correction
    if (server.hasArg("gdb")) gyroDeadband = server.arg("gdb").toFloat(); // Gyro deadband
    if (server.hasArg("ekp")) encKp        = server.arg("ekp").toFloat(); // Encoder Kp
    if (server.hasArg("emc")) encMaxCorr   = server.arg("emc").toInt();   // Encoder max correction
    if (server.hasArg("wkp")) wallKp       = server.arg("wkp").toFloat(); // Wall Kp
    if (server.hasArg("wkd")) wallKd       = server.arg("wkd").toFloat(); // Wall Kd
    if (server.hasArg("wdb")) wallDeadband = server.arg("wdb").toInt();   // Wall deadband
    if (server.hasArg("wmc")) wallMaxCorr  = server.arg("wmc").toInt();   // Wall max correction
    if (server.hasArg("wsl")) wallSensorLimit = server.arg("wsl").toInt(); // Wall sensor limit
    addLog("Corr gKp=" + String(gyroKp) + " eKp=" + String(encKp) + " wKp=" + String(wallKp));
    server.send(200, "text/plain", "OK");
  });

  // ---- /generalcfg ----
  server.on("/generalcfg", []() {
    if (server.hasArg("obs"))  obstacleThreshold = server.arg("obs").toInt();   // Obstacle threshold
    if (server.hasArg("rso"))  rightSensorOffset = server.arg("rso").toInt();   // Right sensor offset
    if (server.hasArg("tsa"))  turnStopAngle     = server.arg("tsa").toFloat(); // Turn stop angle
    if (server.hasArg("ppr"))  PPR               = server.arg("ppr").toFloat(); // Pulses per revolution
    if (server.hasArg("wd")) {
      wheelDiameterMM    = server.arg("wd").toFloat();                           // Wheel diameter
      wheelCircumference = PI * wheelDiameterMM;                                 // Recompute circumference
    }
    if (server.hasArg("tbk"))  turnBackupMM   = server.arg("tbk").toFloat();    // Turn backup distance
    if (server.hasArg("tfwd")) turnForwardMM  = server.arg("tfwd").toFloat();   // Turn forward distance
    if (server.hasArg("bstr")) brakeStrength  = server.arg("bstr").toInt();     // Brake strength
    if (server.hasArg("bdur")) brakeDurationMs = server.arg("bdur").toInt();    // Brake duration
    addLog("Gen obs=" + String(obstacleThreshold) + " tsa=" + String(turnStopAngle));
    server.send(200, "text/plain", "OK");
  });

  // ---- /mazecfg ----
  server.on("/mazecfg", []() {
    if (server.hasArg("nd"))   nodeDistMM     = server.arg("nd").toInt();       // Node distance
    if (server.hasArg("at"))   afterTurnDist  = server.arg("at").toInt();       // After-turn distance

    if (server.hasArg("wdt"))  wallDetectDist = server.arg("wdt").toInt();      // Wall detect distance

      mazeActive = false;                                          // Deactivate maze if active
      resetEncoders(); lockHeading();                              // Reset tracking
      robotState = MOVING_STRAIGHT; addLog("CMD: FORWARD");
    }
    else if (act == "BACKWARD") {                                  // Continuous backward
      mazeActive = false;
      resetEncoders(); lockHeading();
      robotState = MOVING_BACKWARD; addLog("CMD: BACKWARD");
    }
    else if (act == "LEFT_90") {                                   // 90-degree left turn (3-phase)
      mazeActive = false;
      resetEncoders(); robotState = TURN_LEFT_BACKUP; addLog("CMD: LEFT 90");
    }
    else if (act == "RIGHT_90") {                                  // 90-degree right turn (3-phase)
      mazeActive = false;
      resetEncoders(); robotState = TURN_RIGHT_BACKUP; addLog("CMD: RIGHT 90");
    }
    else if (act == "MOVE_DIST") {                                 // Move forward specific distance
      mazeActive = false;
      targetDistanceMM = server.arg("dist").toFloat();             // Get distance from request
      resetEncoders(); lockHeading();
      robotState = MOVING_DISTANCE;
      addLog("CMD: FWD " + String(targetDistanceMM) + "mm");
    }
    else if (act == "BACK_DIST") {                                 // Move backward specific distance
      mazeActive = false;
      targetDistanceMM = server.arg("dist").toFloat();
      resetEncoders(); lockHeading();
      robotState = MOVING_BACKWARD_DISTANCE;
      addLog("CMD: BACK " + String(targetDistanceMM) + "mm");
    }
    else if (act == "TURN_LEFT") {                                 // Pure left turn by angle
      mazeActive = false;
      targetAngleDeg = server.arg("angle").toFloat();              // Get angle
      currentYaw = 0; robotState = TURNING_ANGLE_LEFT;
      addLog("CMD: RAW L " + String(targetAngleDeg) + "deg");
    }
    else if (act == "TURN_RIGHT") {                                // Pure right turn by angle
      mazeActive = false;
      targetAngleDeg = server.arg("angle").toFloat();
      currentYaw = 0; robotState = TURNING_ANGLE_RIGHT;
      addLog("CMD: RAW R " + String(targetAngleDeg) + "deg");
    }
    else if (act == "RESET_ENC") {                                 // Reset encoder counts
      resetEncoders(); addLog("Encoders reset");
    }
    else if (act == "MAZE_RESET") {                                // Reset maze to initial state
      robotState = STOPPED;
      mazeActive = false; stopMotors();
      initMaze();                                                   // Clear maze data
      floodFill();                                                  // Recompute flood
      robotRow = startRow; robotCol = startCol; robotDir = startDir; // Reset position to start
      addLog("MAZE RESET");
    }
    else if (act == "MAZE_START") {                                // Start maze solving
      // Parse start position
      if (server.hasArg("sr")) startRow = constrain(server.arg("sr").toInt(), 0, 7);
      if (server.hasArg("sc")) startCol = constrain(server.arg("sc").toInt(), 0, 7);
      if (server.hasArg("sd")) startDir = constrain(server.arg("sd").toInt(), 0, 3);
      // Reset maze
      stopMotors();
      initMaze();
      floodFill();

      robotRow = startRow;                                          // Set robot position to start
      robotCol = startCol;
      robotDir = startDir;
      mazeActive = true;                                            // Activate maze mode

      // Scan walls at the starting node immediately
      scanMazeWalls();                                               // Detect walls at start cell
      addToPath();                                                   // Record start cell in path
      floodFill();                                                   // Update flood after scan
      addLog("MAZE START (" + String(robotRow) + "," + String(robotCol) +
             ") dir=" + String(robotDir) +
             " w=0x" + String(mazeWalls[robotRow][robotCol], HEX));
      // Check if already at target
      if (isTarget(robotRow, robotCol)) {                           // If start is target area
        robotState = MAZE_DONE;                                      // Maze solved immediately
        addLog("MAZE: Already at target!");
      }
      else {
        // Robot starts at cell center, go straight to deciding next move
        robotState = MAZE_DECIDE;                                    // Move to decision state
        addLog("MZ decide from start");
      }
    }
    server.send(200, "text/plain", "OK");                           // Acknowledge command
  });

  server.begin();                                                   // Start web server
  lastTime = millis();                                              // Initialize timestamp for gyro integration
  addLog("System ready — 8x8 FloodFill");                           // Log ready message
}

// ============================================================
// Main Loop
// ============================================================
void loop() {
  server.handleClient();                                             // Handle incoming web requests
  // ---- Read sensors ----
  if (lidarFront.isRangeComplete()) distFront = lidarFront.readRange(); // Update front distance if available
  if (lidarLeft.isRangeComplete())  distLeft  = lidarLeft.readRange();  // Update left distance
  if (lidarRight.isRangeComplete()) distRight = max(0, lidarRight.readRange() - rightSensorOffset); // Update right with offset

  sensors_event_t a, g, temp;                                        // Create sensor event structures
  mpu.getEvent(&a, &g, &temp);                                       // Read MPU6050 data

  unsigned long now = millis();                                      // Current time in ms
  float dt = (now - lastTime) / 1000.0f;                             // Time delta in seconds
  lastTime = now;                                                     // Update last time

  gyroX     = g.gyro.x * 57.2958f;                                   // Convert gyro X to deg/s (rad/s * 180/pi)
  gyroY     = g.gyro.y * 57.2958f;                                   // Gyro Y
  gyroZ_raw = g.gyro.z * 57.2958f;                                   // Gyro Z (yaw rate)
  accelX    = a.acceleration.x / 9.81f;                              // Convert accelerometer X to g (m/s² / 9.81)
  accelY    = a.acceleration.y / 9.81f;                              // Accelerometer Y
  accelZ    = a.acceleration.z / 9.81f;                              // Accelerometer Z

  if (abs(gyroZ_raw) > gyroDeadband) currentYaw += gyroZ_raw * dt;   // Integrate yaw rate if above deadband

  // ---- State machine ----
  switch (robotState) {                                               // Handle current state

    case STOPPED:                                                     // Robot stopped
      setMotors(0, 0);                                                // Ensure motors off
      break;

    // =============== MANUAL MOVEMENT ===============
    case MOVING_STRAIGHT:                                             // Continuous forward movement
      if (distFront > 0 && distFront < obstacleThreshold) {          // If obstacle ahead
        robotState = STOPPED;                                         // Stop
        stopMotors();                                                  // Apply brakes
        addLog("STOP: obstacle " + String(distFront) + "mm");        // Log
      } else {
        driveFused(moveSpeed);                                        // Otherwise drive forward with correction
      }
      break;

    case MOVING_DISTANCE:                                             // Move forward a specific distance
      if (distFront > 0 && distFront < obstacleThreshold) {          // Obstacle detected
        robotState = STOPPED;
        stopMotors();
        addLog("STOP: obstacle " + String(distFront) + "mm");
      } else if (encoderAvgDistanceMM() >= targetDistanceMM) {       // Reached target distance
        robotState = STOPPED;
        stopMotors();
        addLog("DONE: " + String(encoderAvgDistanceMM(), 1) + "mm");
      } else {
        driveFused(moveSpeed);                                        // Keep moving
      }
      break;

    case MOVING_BACKWARD:                                             // Continuous backward movement
      driveBackwardFused(moveSpeed);                                  // Drive backward with correction
      break;
    case MOVING_BACKWARD_DISTANCE:                                    // Backward a specific distance
      if (encoderAvgDistanceMM() >= targetDistanceMM) {              // Reached target
        robotState = STOPPED;
        stopMotors();
        addLog("DONE back: " + String(encoderAvgDistanceMM(), 1) + "mm");
      } else {
        driveBackwardFused(moveSpeed);                                // Keep moving backward
      }
      break;

    // =============== MANUAL 3-PHASE TURNS ===============
    case TURN_LEFT_BACKUP:                                            // Phase 1: backup before left turn
      if (encoderAvgDistanceMM() >= turnBackupMM) {                   // Backup distance reached
        resetEncoders();                                               // Reset encoders for next phase
        currentYaw = 0;                                                // Reset yaw for pivot
        robotState = TURN_LEFT_PIVOT;                                 // Move to pivot phase
      } else {
        setMotors(-moveSpeed, -moveSpeed);                            // Drive backward
      }
      break;

    case TURN_LEFT_PIVOT:                                             // Phase 2: pivot left
      if (abs(currentYaw) >= turnStopAngle) {                         // Reached target angle
        stopMotors();                                                  // Stop
        delay(80);                                                     // Brief pause
        resetEncoders(); lockHeading();                                // Reset for forward phase
        robotState = TURN_LEFT_FORWARD;                               // Move to forward phase
      } else {
        setMotors(0, turnSpeed);                                      // Turn left: right motor forward, left stopped
      }
      break;

    case TURN_LEFT_FORWARD:                                           // Phase 3: move forward after left turn
      if (encoderAvgDistanceMM() >= turnForwardMM) {                  // Forward distance reached
        robotState = STOPPED;                                          // Stop
        stopMotors();
        addLog("TL DONE");
      } else {
        driveFused(moveSpeed);                                         // Drive forward with correction
      }
      break;

    case TURN_RIGHT_BACKUP:                                           // Phase 1: backup before right turn
      if (encoderAvgDistanceMM() >= turnBackupMM) {
        resetEncoders();
        currentYaw = 0;
        robotState = TURN_RIGHT_PIVOT;
      } else {
        setMotors(-moveSpeed, -moveSpeed);
      }
      break;

    case TURN_RIGHT_PIVOT:                                            // Phase 2: pivot right
      if (abs(currentYaw) >= turnStopAngle) {
        stopMotors();
        delay(80);
        resetEncoders(); lockHeading();
        robotState = TURN_RIGHT_FORWARD;
      } else {
        setMotors(turnSpeed, 0);                                      // Turn right: left motor forward, right stopped
      }
      break;

    case TURN_RIGHT_FORWARD:                                          // Phase 3: forward after right turn
      if (encoderAvgDistanceMM() >= turnForwardMM) {
        robotState = STOPPED;
        stopMotors();
        addLog("TR DONE");
      } else {
        driveFused(moveSpeed);
      }
      break;

    // =============== RAW ANGLE TURNS ===============
    case TURNING_ANGLE_LEFT:                                          // Pure left turn by angle
      if (abs(currentYaw) >= targetAngleDeg) {                        // Target angle reached
        robotState = STOPPED;
        stopMotors();
        addLog("RAW L done yaw=" + String(currentYaw, 1));
      } else { setMotors(turnSpeed, 0); }                             // Turn left (left motor forward, right stopped)
      break;
    case TURNING_ANGLE_RIGHT:                                         // Pure right turn by angle
      if (abs(currentYaw) >= targetAngleDeg) {
        robotState = STOPPED;
        stopMotors();
        addLog("RAW R done yaw=" + String(currentYaw, 1));
      } else { setMotors(0, turnSpeed); }                             // Turn right
      break;

    // =============== MAZE SOLVING STATES ===============

    case MAZE_SCAN:                                                   // Pause and scan walls at cell
      // Pause for stable sensor readings, then scan walls
      if (mazePauseStart > 0 && (millis() - mazePauseStart) < (unsigned long)mazePauseMs) { // If pause time not elapsed
        setMotors(0, 0);                                              // Stay stopped
        // Hold still
        break;
      }
      mazePauseStart = 0;                                             // Reset pause timer
      // Scan walls using current LiDAR readings
      scanMazeWalls();                                                 // Detect and record walls
      addToPath();                                                     // Record current cell in path
      addLog("MZ scan (" + String(robotRow) + "," + String(robotCol) +
             ") w=0x" + String(mazeWalls[robotRow][robotCol], HEX)); // Log walls found
      robotState = MAZE_DECIDE;                                       // Move to decision state
      break;

    case MAZE_DECIDE: {                                               // Decide next move based on flood fill
      // Check if we reached the target
      if (isTarget(robotRow, robotCol)) {                             // If at target area
        robotState = MAZE_DONE;                                        // Maze solved
        stopMotors();
        mazeActive = false;                                            // Deactivate maze mode
        addLog("*** MAZE SOLVED! ***");
        break;
      }

      // Run floodfill
      floodFill();                                                     // Update flood values
      // Find best direction
      int bestDir = findBestDirection();                              // Get next direction (0-3) or -1 if none
      if (bestDir == -1 || mazeFlood[robotRow][robotCol] == 255) {   // If no valid direction or unreachable
        robotState = STOPPED;                                          // Stop and abort
        stopMotors();
        mazeActive = false;
        addLog("MZ: No path to target!");
        break;
      }

      mazeNextDir = bestDir;                                           // Store next direction
      int turnDiff = (mazeNextDir - robotDir + 4) % 4;                // Compute turn difference (0=straight,1=right,2=back,3=left)
      addLog("MZ decide: go dir " + String(mazeNextDir) +
             " turn=" + String(turnDiff) +
             " flood=" + String(mazeFlood[robotRow + DR[bestDir]][robotCol + DC[bestDir]])); // Log decision
      if (turnDiff == 0) {                                             // Straight ahead
        // Straight — go forward directly
        mazeMoveDist = nodeDistMM;                                    // Set move distance to full cell
        mazeFromReverse = false;                                      // Not reversing
        resetEncoders(); lockHeading();                               // Reset tracking
        robotState = MAZE_FORWARD;                                    // Move forward state
      } else if (turnDiff == 2) {                                     // 180-degree turn required
        // 180-degree turnaround required
        int nextR = robotRow + DR[bestDir];                           // Cell we would move into if we turned
        int nextC = robotCol + DC[bestDir];

        if (mazeVisited[nextR][nextC]) {                              // If that cell is already visited
          // Path is KNOWN: Safe to reverse without turning
          mazeMoveDist = nodeDistMM;                                   // Reverse one cell distance
          resetEncoders(); lockHeading();                             // Reset
          robotState = MAZE_REVERSE;                                  // Go to reverse state
          addLog("MZ reverse (known)");
        } else {
          // Path is UNKNOWN: Must face forward to scan, do a physical 180
          mazeTurnDir = 1; // 1 = Right spin                       // Turn right for 180 (or left; using right here)
          mazePivotAngle = turn180Angle; // Uses the 170-180 deg target from config
          resetEncoders();                                            // Reset encoders
          robotState = MAZE_TURN_BACKUP; // Backs up first to clear the tail! // Start backup before turn
          addLog("MZ 180 pivot (unknown)");
        }
      } else {
        // Need to turn 90 degrees
        if (turnDiff == 1) {       // Right 90
          mazeTurnDir = 1;                                            // Right turn
          mazePivotAngle = turnStopAngle;                             // Use configured 90° angle
        } else {                    // Left 90 (turnDiff == 3)
          mazeTurnDir = -1;                                           // Left turn
          mazePivotAngle = turnStopAngle;
        }
        resetEncoders();                                              // Reset for turn
        robotState = MAZE_TURN_BACKUP;                                // Start backup phase
      }
      break;
    }

    case MAZE_TURN_BACKUP:                                            // Backup before turn in maze
      if (encoderAvgDistanceMM() >= mazeTurnBackupMM) {               // Backup distance reached
        resetEncoders();                                               // Reset for pivot
        currentYaw = 0;                                                // Reset yaw for angle measurement
        robotState = MAZE_TURN_PIVOT;                                 // Move to pivot phase
        addLog("MZ pivot start dir=" + String(mazeTurnDir));
      } else {
        setMotors(-moveSpeed, -moveSpeed);                            // Drive backward
      }
      break;
    case MAZE_TURN_PIVOT:                                             // Pivot turn
      if (abs(currentYaw) >= mazePivotAngle) {                        // Reached target angle
        stopMotors();                                                  // Stop
        delay(80);                                                     // Brief pause
        resetEncoders(); lockHeading();                               // Reset for forward phase
        robotState = MAZE_TURN_ADVANCE;                               // Move to advance phase
        addLog("MZ pivot done yaw=" + String(currentYaw, 1));
      } else {
        if (mazeTurnDir == 1) {                                       // Right turn
          setMotors(0, turnSpeed);                                     // Right motor forward (or left?)
          // Right turn
        } else {                                                       // Left turn
          setMotors(turnSpeed, 0);                                     // Left motor forward
          // Left turn
        }
      }
      break;
    case MAZE_TURN_ADVANCE:                                           // Move forward after pivot
      if (encoderAvgDistanceMM() >= turnForwardMM) {                  // Reached forward distance
        // Update robot direction
        robotDir = mazeNextDir;                                        // Set new direction
        addLog("MZ advance done, dir=" + String(robotDir));
        // Back up to center on node if configured
        if (mazeTurnCenterMM > 0) {                                   // If extra centering backup needed
          resetEncoders();
          robotState = MAZE_TURN_CENTER;                               // Go to centering state
        } else {
          // After reverse the robot is at cell center, needs full cell distance
          mazeMoveDist = mazeFromReverse ? nodeDistMM : afterTurnDist; // Set move distance: if from reverse, full cell; else after-turn distance
          mazeFromReverse = false;                                     // Reset flag
          if (mazeMoveDist < 10) mazeMoveDist = 10;                   // Ensure minimum distance
          resetEncoders(); lockHeading();
          robotState = MAZE_FORWARD;                                   // Move forward
          addLog("MZ fwd=" + String(mazeMoveDist));
        }
      } else {
        driveFused(moveSpeed);                                         // Drive forward with correction
      }
      break;

    case MAZE_TURN_CENTER:                                            // Backup to center after turn
      if (encoderAvgDistanceMM() >= mazeTurnCenterMM) {               // Centering distance reached
        stopMotors();
        delay(50);
        // After reverse the robot is at cell center, needs full cell distance
        mazeMoveDist = mazeFromReverse ? nodeDistMM : afterTurnDist;   // Set move distance
        mazeFromReverse = false;
        if (mazeMoveDist < 10) mazeMoveDist = 10;
        resetEncoders(); lockHeading();
        robotState = MAZE_FORWARD;                                     // Move forward
        addLog("MZ centered, fwd=" + String(mazeMoveDist));
      } else {
        setMotors(-moveSpeed, -moveSpeed);                             // Drive backward
      }
      break;

    case MAZE_FORWARD:                                                 // Move forward one cell in maze
      if (encoderAvgDistanceMM() >= mazeMoveDist) {                    // Reached target distance
        // Arrived at distance target
        stopMotors();
        // Update position to next cell
        robotRow += DR[robotDir];                                       // Move row according to direction
        robotCol += DC[robotDir];                                       // Move column
        robotRow = constrain(robotRow, 0, 7);                           // Keep within bounds
        robotCol = constrain(robotCol, 0, 7);
        addLog("MZ arrived (" + String(robotRow) + "," + String(robotCol) + ")");
        // Go to scan
        mazePauseStart = millis();                                      // Start pause timer for scanning
        robotState = MAZE_SCAN;                                         // Move to scan state
      }
      else if (distFront > 0 && distFront < frontStopDist) {           // Obstacle detected mid-move
        // Front obstacle detected mid-move! Emergency handling
        stopMotors();
        addLog("MZ front wall mid-move! d=" + String(distFront));
        // Mark front wall at current position
        setWallBilateral(robotRow, robotCol, robotDir);                 // Record wall in front
        // Stay at current cell, re-decide
        mazePauseStart = millis();                                      // Start pause for scan
        robotState = MAZE_SCAN;                                         // Go back to scan (will re-decide)
      }
      else {
        driveFused(moveSpeed);                                          // Continue moving with correction
      }
      break;

    case MAZE_REVERSE:                                                 // Move backward one cell
      // Drive backward one cell distance
      if (encoderAvgDistanceMM() >= mazeMoveDist) {                    // Reached distance
        stopMotors();
        // Update position: move in the OPPOSITE of robotDir (going backward)
        int backDir = (robotDir + 2) % 4;                              // Opposite direction
        robotRow += DR[backDir];                                        // Update row
        robotCol += DC[backDir];                                        // Update column
        robotRow = constrain(robotRow, 0, 7);
        robotCol = constrain(robotCol, 0, 7);
        // DO NOT flip robotDir — we're still facing the same way, just backing up
        addLog("MZ reversed to (" + String(robotRow) + "," + String(robotCol) + ")");
        addToPath();                                                    // Record new cell in path

        // Scan walls at this cell and re-run floodfill
        scanMazeWalls();                                                // Detect walls at new cell
        floodFill();                                                    // Update flood values
        // Check if a non-180 route exists from here
        int checkBest = findBestDirection();                           // Find next best direction
        if (checkBest == -1 || mazeFlood[robotRow][robotCol] == 255) { // No valid path
          robotState = STOPPED;
          stopMotors();
          mazeActive = false;
          addLog("MZ: No path to target!");
        } else {
          int checkTurn = (checkBest - robotDir + 4) % 4;              // Compute turn needed
          if (checkTurn == 2) {                                         // Still needs 180
            // Still needs 180 — keep reversing
            mazeMoveDist = nodeDistMM;                                  // Set reverse distance again
            resetEncoders(); lockHeading();
            addLog("MZ keep reversing");
            // stay in MAZE_REVERSE (state unchanged)
          } else {
            // Found a non-180 route — pause, scan properly, then decide
            mazeFromReverse = true;                                     // Indicate we came from reverse (for distance handling)
            mazePauseStart = millis();                                  // Pause before scanning
            robotState = MAZE_DECIDE;                                   // Go to decision (will pause and scan)
            addLog("MZ reverse done, can go dir=" + String(checkBest));
          }
        }
      } else {
        driveBackwardFused(moveSpeed);                                 // Continue reversing with correction
      }
      break;

    case MAZE_DONE:                                                    // Maze solved
      setMotors(0, 0);                                                  // Stop motors
      // Stay here until user stops or resets
      break;
  }
} // End of loop
