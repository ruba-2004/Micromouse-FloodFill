void IRAM_ATTR readLeftEncoder() {
  if (digitalRead(ENC_LEFT_A) == digitalRead(ENC_LEFT_B)) leftEncoderTicks++;
  else leftEncoderTicks--;
}
void IRAM_ATTR readRightEncoder() {
  if (digitalRead(ENC_RIGHT_A) == digitalRead(ENC_RIGHT_B)) rightEncoderTicks++;
  else rightEncoderTicks--;
}

// ============================================================
// Motor control
// ============================================================
void setMotors(int L, int R) {
  L = constrain(L, -255, 255);
  R = constrain(R, -255, 255);
  if      (L > 0) { analogWrite(IN1, L);   analogWrite(IN2, 0); }
  else if (L < 0) { analogWrite(IN1, 0);   analogWrite(IN2, -L); }
  else            { analogWrite(IN1, 0);   analogWrite(IN2, 0); }
  if      (R > 0) { analogWrite(IN3, R);   analogWrite(IN4, 0); }
  else if (R < 0) { analogWrite(IN3, 0);   analogWrite(IN4, -R); }
  else            { analogWrite(IN3, 0);   analogWrite(IN4, 0); }
}

void stopMotors() {
  setMotors(-brakeStrength, -brakeStrength);
  delay(brakeDurationMs);
  setMotors(0, 0);
}

float encoderAvgDistanceMM() {
  float dL = (abs((float)leftEncoderTicks)  / PPR) * wheelCircumference;
  float dR = (abs((float)rightEncoderTicks) / PPR) * wheelCircumference;
  return (dL + dR) / 2.0f;
}

void resetEncoders() {

  rightEncoderTicks = 0;
}

// ============================================================
// 3-SENSOR FUSION STRAIGHT-LINE CORRECTION
// ============================================================
int fusedStraightCorrectionFull() {
  // Layer 1: Gyro
  float gyroErr   = currentYaw - headingLock;
  float gyroDeriv = gyroErr - lastGyroErr;
  lastGyroErr     = gyroErr;
  float gyroCorr  = -(gyroKp * gyroErr + gyroKd * gyroDeriv);
  gyroCorr = constrain(gyroCorr, -(float)gyroMaxCorr, (float)gyroMaxCorr);
  live_fc_gyro = gyroCorr;

  // Layer 2: Encoder
  long relLeft  = leftEncoderTicks  - encLeftStart;
  long relRight = rightEncoderTicks - encRightStart;
  long encErr   = relLeft - relRight;
  float encCorr = encKp * (float)encErr;
  encCorr = constrain(encCorr, -(float)encMaxCorr, (float)encMaxCorr);
  live_fc_enc = encCorr;

  // Layer 3: LiDAR (Absolute Single-Wall Tracking)
  float wallCorr = 0.0f;
  bool hasLeft  = (distLeft  > 0 && distLeft  < wallSensorLimit);
  bool hasRight = (distRight > 0 && distRight < wallSensorLimit);
  bool wallsValid = (hasLeft || hasRight); 

  // GEOMETRY CALCULATION:
  // Node is 200mm wide. Robot is 100mm wide. 
  // (200 - 100) / 2 = 50mm distance to each wall.
  // Adjust this slightly if your sensors are mounted further inside the chassis.
  float idealWallDist = 50.0f; 
  int wallErr = 0;

  if (hasLeft && hasRight) {
    // Both walls: normal centering
    wallErr = distRight - distLeft;
  } else if (hasLeft && !hasRight) {
    // Left wall only: push towards/away from left wall to maintain ideal distance
    // Multiplied by 2 to match the error magnitude of the two-wall state
    wallErr = (idealWallDist - distLeft) * 2; 
  } else if (!hasLeft && hasRight) {
    // Right wall only
    wallErr = (distRight - idealWallDist) * 2;
  }

  if (wallsValid) {
    if (abs(wallErr) > wallDeadband) {
      int wallDeriv = wallErr - lastWallErr;
      wallCorr = wallKp * wallErr + wallKd * (float)wallDeriv;
      wallCorr = constrain(wallCorr, -(float)wallMaxCorr, (float)wallMaxCorr);
    }
    lastWallErr = wallErr;
  } else {
    lastWallErr = 0;
  }
  live_fc_lidar = wallCorr;

  // Weights
  float wG = wGyro, wE = wEncoder, wL = wLidar;
  if (!wallsValid) {
    float r = wLidar;
    wG = wGyro    + r * (wGyro    / (wGyro + wEncoder));
    wE = wEncoder + r * (wEncoder / (wGyro + wEncoder));
    wL = 0.0f;
  }

  float fused = (wG * gyroCorr) + (wE * encCorr) + (wL * wallCorr);
  fused = constrain(fused, -(float)fusedMaxCorr, (float)fusedMaxCorr);
  live_fc_fused = fused;
  return (int)fused;
}

void driveFused(int spd) {
  int corr = fusedStraightCorrectionFull();
  setMotors(constrain(spd - corr, spd - fusedMaxCorr, spd + fusedMaxCorr),
            constrain(spd + corr, spd - fusedMaxCorr, spd + fusedMaxCorr));
}

void driveBackwardFused(int spd) {
  int corr = fusedStraightCorrectionFull();
  setMotors(constrain(-spd - corr, -(spd + fusedMaxCorr), -(spd - fusedMaxCorr)),
            constrain(-spd + corr, -(spd + fusedMaxCorr), -(spd - fusedMaxCorr)));
}

void lockHeading() {
  headingLock   = currentYaw;
  lastGyroErr   = 0.0f;
  lastWallErr   = 0;
  encLeftStart  = leftEncoderTicks;
  encRightStart = rightEncoderTicks;
  live_fc_gyro = live_fc_enc = live_fc_lidar = live_fc_fused = 0;
}

bool isTarget(int r, int c) {
  return (r == 3 || r == 4) && (c == 3 || c == 4);
}

void setWallBilateral(int r, int c, int d) {
  if (r < 0 || r >= 8 || c < 0 || c >= 8) return;
  mazeWalls[r][c] |= (1 << d);
  int nr = r + DR[d];
  int nc = c + DC[d];
  if (nr >= 0 && nr < 8 && nc >= 0 && nc < 8) {
    mazeWalls[nr][nc] |= (1 << ((d + 2) % 4));
  }
}

void initMaze() {
  memset(mazeWalls, 0, sizeof(mazeWalls));
  memset(mazeFlood, 255, sizeof(mazeFlood));
  memset(mazeVisited, 0, sizeof(mazeVisited));
  pathLen = 0;
  // Outer boundary walls
  for (int i = 0; i < 8; i++) {
    mazeWalls[0][i] |= WALL_N;
    mazeWalls[7][i] |= WALL_S;
    mazeWalls[i][0] |= WALL_W;
    mazeWalls[i][7] |= WALL_E;
  }
}

void floodFill() {
  memset(mazeFlood, 255, sizeof(mazeFlood));
  // BFS queue
  uint8_t qr[64], qc[64];
  int front = 0, back = 0;
  // Seed: center 4 target cells
  for (int r = 3; r <= 4; r++) {
    for (int c = 3; c <= 4; c++) {
      mazeFlood[r][c] = 0;
      qr[back] = r; qc[back] = c; back++;
    }
  }

  while (front < back) {
    int r = qr[front];
    int c = qc[front]; front++;
    uint8_t nextVal = mazeFlood[r][c] + 1;
    for (int d = 0; d < 4; d++) {
      if (mazeWalls[r][c] & (1 << d)) continue;
      // wall blocks
      int nr = r + DR[d];
      int nc = c + DC[d];
      if (nr < 0 || nr >= 8 || nc < 0 || nc >= 8) continue;
      if (mazeFlood[nr][nc] > nextVal) {
        mazeFlood[nr][nc] = nextVal;
        qr[back] = nr;
        qc[back] = nc; back++;
      }
    }
  }
}

void scanMazeWalls() {
  int frontDir = robotDir;
  int rightDir = (robotDir + 1) % 4;
  int leftDir  = (robotDir + 3) % 4;
  // Take multiple readings for reliability
  const int NUM_SAMPLES = 5;
  int frontHits = 0, leftHits = 0, rightHits = 0;
  for (int s = 0; s < NUM_SAMPLES; s++) {
    // Read fresh sensor data
    if (lidarFront.isRangeComplete()) distFront = lidarFront.readRange();
    if (lidarLeft.isRangeComplete())  distLeft  = lidarLeft.readRange();
    if (lidarRight.isRangeComplete()) distRight = max(0, lidarRight.readRange() - rightSensorOffset);
    if (distFront > 0 && distFront < wallDetectDist) frontHits++;
    if (distLeft  > 0 && distLeft  < wallDetectDist) leftHits++;
    if (distRight > 0 && distRight < wallDetectDist) rightHits++;

    delay(20);
    // small delay between samples
  }

  // Majority vote: only mark wall if 3+ out of 5 samples agree
  int threshold = (NUM_SAMPLES / 2) + 1; // 3

  if (frontHits >= threshold)
    setWallBilateral(robotRow, robotCol, frontDir);
  if (rightHits >= threshold)
    setWallBilateral(robotRow, robotCol, rightDir);
  if (leftHits >= threshold)
    setWallBilateral(robotRow, robotCol, leftDir);

  addLog("MZ walls F:" + String(frontHits) + " L:" + String(leftHits) +
         " R:" + String(rightHits) + "/" + String(NUM_SAMPLES));
  mazeVisited[robotRow][robotCol] = true;
}

void addToPath() {
  if (pathLen < 256) {
    pathR[pathLen] = robotRow;
    pathC[pathLen] = robotCol;
    pathLen++;
  }
}

// Find the best direction to move from current cell (lowest flood value neighbor)
// Priority on tie: straight > right > left > back
int findBestDirection() {
  int bestDir = -1;
  int bestFlood = 255;

  int checkOrder[4];
  checkOrder[0] = robotDir;               // straight
  checkOrder[1] = (robotDir + 1) % 4;     // right
  checkOrder[2] = (robotDir + 3) % 4;     // left
  checkOrder[3] = (robotDir + 2) % 4;     // back

  for (int i = 0; i < 4; i++) {
    int d = checkOrder[i];
    if (mazeWalls[robotRow][robotCol] & (1 << d)) continue;
    int nr = robotRow + DR[d];
    int nc = robotCol + DC[d];
    if (nr < 0 || nr >= 8 || nc < 0 || nc >= 8) continue;
    if (mazeFlood[nr][nc] < bestFlood) {
      bestFlood = mazeFlood[nr][nc];
      bestDir = d;
    }
  }
  return bestDir;
}

    case MAZE_SCAN:          return "SCANNING";

// Setup

void setup() {
  Serial.begin(115200);

  pinMode(ENC_LEFT_A,  INPUT); pinMode(ENC_LEFT_B,  INPUT);
  pinMode(ENC_RIGHT_A, INPUT); pinMode(ENC_RIGHT_B, INPUT);
  attachInterrupt(digitalPinToInterrupt(ENC_LEFT_A),  readLeftEncoder,  CHANGE);
  attachInterrupt(digitalPinToInterrupt(ENC_RIGHT_A), readRightEncoder, CHANGE);

  pinMode(IN1, OUTPUT); pinMode(IN2, OUTPUT);
  pinMode(IN3, OUTPUT); pinMode(IN4, OUTPUT);
  setMotors(0, 0);
  pinMode(XSHUT_LEFT,  OUTPUT);
  pinMode(XSHUT_FRONT, OUTPUT);
  pinMode(XSHUT_RIGHT, OUTPUT);
  digitalWrite(XSHUT_LEFT,  LOW);
  digitalWrite(XSHUT_FRONT, LOW);
  digitalWrite(XSHUT_RIGHT, LOW);
  delay(10);

  Wire.begin(GYRO_SDA,  GYRO_SCL);
  Wire1.begin(LIDAR_SDA, LIDAR_SCL);
  if (!mpu.begin(0x68, &Wire)) {
    addLog("ERR: MPU6050");
  } else {
    mpu.setGyroRange(MPU6050_RANGE_500_DEG);
    mpu.setFilterBandwidth(MPU6050_BAND_21_HZ);
    addLog("MPU6050 OK");
  }

  digitalWrite(XSHUT_LEFT,  HIGH); delay(10);
  if (!lidarLeft.begin(0x30,  false, &Wire1)) addLog("ERR: LiDAR L"); else addLog("LiDAR L OK");
  digitalWrite(XSHUT_FRONT, HIGH);
  delay(10);
  if (!lidarFront.begin(0x31, false, &Wire1)) addLog("ERR: LiDAR F"); else addLog("LiDAR F OK");
  digitalWrite(XSHUT_RIGHT, HIGH); delay(10);
  if (!lidarRight.begin(0x29, false, &Wire1)) addLog("ERR: LiDAR R"); else addLog("LiDAR R OK");

  lidarLeft.startRangeContinuous();
  lidarFront.startRangeContinuous();
  lidarRight.startRangeContinuous();

  int att = 0;
  while (WiFi.status() != WL_CONNECTED && att < 20) {

      if (server.hasArg("sc")) startCol = constrain(server.arg("sc").toInt(), 0, 7);

             " w=0x" + String(mazeWalls[robotRow][robotCol], HEX));

        robotState = MAZE_DECIDE;
        addLog("MZ decide from start");
      }
    }
    server.send(200, "text/plain", "OK");
  });

  addLog("System ready — 8x8 FloodFill");

// ============================================================
// Main Loop
// ============================================================
void loop() {
  server.handleClient();
  // ---- Read sensors ----
  if (lidarFront.isRangeComplete()) distFront = lidarFront.readRange();
  if (lidarLeft.isRangeComplete())  distLeft  = lidarLeft.readRange();
  if (lidarRight.isRangeComplete()) distRight = max(0, lidarRight.readRange() - rightSensorOffset);

  sensors_event_t a, g, temp;
  mpu.getEvent(&a, &g, &temp);

  unsigned long now = millis();
  float dt = (now - lastTime) / 1000.0f;
  lastTime = now;

  gyroX     = g.gyro.x * 57.2958f;
  gyroY     = g.gyro.y * 57.2958f;
  gyroZ_raw = g.gyro.z * 57.2958f;
  accelX    = a.acceleration.x / 9.81f;
  accelY    = a.acceleration.y / 9.81f;
  accelZ    = a.acceleration.z / 9.81f;

  if (abs(gyroZ_raw) > gyroDeadband) currentYaw += gyroZ_raw * dt;
  // ---- State machine ----
  switch (robotState) {

    case STOPPED:
      setMotors(0, 0);
      break;

    // =============== MANUAL MOVEMENT ===============
    case MOVING_STRAIGHT:
      if (distFront > 0 && distFront < obstacleThreshold) {
        robotState = STOPPED;
        stopMotors();
        addLog("STOP: obstacle " + String(distFront) + "mm");
      } else {
        driveFused(moveSpeed);
      }
      break;

    case MOVING_DISTANCE:
      if (distFront > 0 && distFront < obstacleThreshold) {
        robotState = STOPPED;
        stopMotors();
        addLog("STOP: obstacle " + String(distFront) + "mm");
      } else if (encoderAvgDistanceMM() >= targetDistanceMM) {
        robotState = STOPPED;
        stopMotors();
        addLog("DONE: " + String(encoderAvgDistanceMM(), 1) + "mm");
      } else {
        driveFused(moveSpeed);
      }
      break;

    case MOVING_BACKWARD:
      driveBackwardFused(moveSpeed);
      break;
    case MOVING_BACKWARD_DISTANCE:
      if (encoderAvgDistanceMM() >= targetDistanceMM) {
        robotState = STOPPED;
        stopMotors();
        addLog("DONE back: " + String(encoderAvgDistanceMM(), 1) + "mm");
      } else {
        driveBackwardFused(moveSpeed);
      }
      break;

    // =============== MANUAL 3-PHASE TURNS ===============
    case TURN_LEFT_BACKUP:
      if (encoderAvgDistanceMM() >= turnBackupMM) {
        resetEncoders();
        currentYaw = 0;
        robotState = TURN_LEFT_PIVOT;
      } else {
        setMotors(-moveSpeed, -moveSpeed);
      }
      break;

    case TURN_LEFT_PIVOT:
      if (abs(currentYaw) >= turnStopAngle) {
        stopMotors();
        delay(80);
        resetEncoders(); lockHeading();
        robotState = TURN_LEFT_FORWARD;
      } else {
        setMotors(0, turnSpeed);
      }
      break;

    case TURN_LEFT_FORWARD:
      if (encoderAvgDistanceMM() >= turnForwardMM) {
        robotState = STOPPED;
        stopMotors();
        addLog("TL DONE");
      } else {
        driveFused(moveSpeed);
      }
      break;

    case TURN_RIGHT_BACKUP:
      if (encoderAvgDistanceMM() >= turnBackupMM) {
        resetEncoders();
        currentYaw = 0;
        robotState = TURN_RIGHT_PIVOT;
      } else {
        setMotors(-moveSpeed, -moveSpeed);
      }
      break;

    case TURN_RIGHT_PIVOT:
      if (abs(currentYaw) >= turnStopAngle) {
        stopMotors();
        delay(80);
        resetEncoders(); lockHeading();
        robotState = TURN_RIGHT_FORWARD;
      } else {
        setMotors(turnSpeed, 0);
      }
      break;

    case TURN_RIGHT_FORWARD:
      if (encoderAvgDistanceMM() >= turnForwardMM) {
        robotState = STOPPED;
        stopMotors();
        addLog("TR DONE");
      } else {
        driveFused(moveSpeed);
      }
      break;

    // =============== RAW ANGLE TURNS ===============
    case TURNING_ANGLE_LEFT:
      if (abs(currentYaw) >= targetAngleDeg) {
        robotState = STOPPED;
        stopMotors();
        addLog("RAW L done yaw=" + String(currentYaw, 1));
      } else { setMotors(turnSpeed, 0); }
      break;
    case TURNING_ANGLE_RIGHT:
      if (abs(currentYaw) >= targetAngleDeg) {
        robotState = STOPPED;
        stopMotors();
        addLog("RAW R done yaw=" + String(currentYaw, 1));
      } else { setMotors(0, turnSpeed); }
      break;
    // =============== MAZE SOLVING STATES ===============

    case MAZE_SCAN:
      // Pause for stable sensor readings, then scan walls
      if (mazePauseStart > 0 && (millis() - mazePauseStart) < (unsigned long)mazePauseMs) {
        setMotors(0, 0);
        // Hold still
        break;
      }
      mazePauseStart = 0;
      // Scan walls using current LiDAR readings
      scanMazeWalls();
      addToPath();
      addLog("MZ scan (" + String(robotRow) + "," + String(robotCol) +
             ") w=0x" + String(mazeWalls[robotRow][robotCol], HEX));
      robotState = MAZE_DECIDE;
      break;

    case MAZE_DECIDE: {
      // Check if we reached the target
      if (isTarget(robotRow, robotCol)) {
        robotState = MAZE_DONE;
        stopMotors();
        mazeActive = false;
        addLog("*** MAZE SOLVED! ***");
        break;
      }

      // Run floodfill
      floodFill();
      // Find best direction
      int bestDir = findBestDirection();
      if (bestDir == -1 || mazeFlood[robotRow][robotCol] == 255) {
        robotState = STOPPED;
        stopMotors();
        mazeActive = false;
        addLog("MZ: No path to target!");
        break;
      }

      mazeNextDir = bestDir;
      int turnDiff = (mazeNextDir - robotDir + 4) % 4;
      addLog("MZ decide: go dir " + String(mazeNextDir) +
             " turn=" + String(turnDiff) +
             " flood=" + String(mazeFlood[robotRow + DR[bestDir]][robotCol + DC[bestDir]]));
      if (turnDiff == 0) {
        // Straight — go forward directly
        mazeMoveDist = nodeDistMM;
        mazeFromReverse = false;
        resetEncoders(); lockHeading();
        robotState = MAZE_FORWARD;
      } else if (turnDiff == 2) {
        // 180-degree turnaround required
        int nextR = robotRow + DR[bestDir];
        int nextC = robotCol + DC[bestDir];

        if (mazeVisited[nextR][nextC]) {
          // Path is KNOWN: Safe to reverse without turning
          mazeMoveDist = nodeDistMM;
          resetEncoders(); lockHeading();
          robotState = MAZE_REVERSE;
          addLog("MZ reverse (known)");
        } else {
          // Path is UNKNOWN: Must face forward to scan, do a physical 180
          mazeTurnDir = 1; // 1 = Right spin
          mazePivotAngle = turn180Angle; // Uses the 170-180 deg target from config
          resetEncoders();
          robotState = MAZE_TURN_BACKUP; // Backs up first to clear the tail!
          addLog("MZ 180 pivot (unknown)");
        }
      } else {
        // Need to turn
        if (turnDiff == 1) {       // Right 90
          mazeTurnDir = 1;
          mazePivotAngle = turnStopAngle;
        } else {                    // Left 90 (turnDiff == 3)
          mazeTurnDir = -1;
          mazePivotAngle = turnStopAngle;
        }
        resetEncoders();
        robotState = MAZE_TURN_BACKUP;
      }
      break;
    }

    case MAZE_TURN_BACKUP:
      if (encoderAvgDistanceMM() >= mazeTurnBackupMM) {
        resetEncoders();
        currentYaw = 0;
        robotState = MAZE_TURN_PIVOT;
        addLog("MZ pivot start dir=" + String(mazeTurnDir));
      } else {
        setMotors(-moveSpeed, -moveSpeed);
      }
      break;
    case MAZE_TURN_PIVOT:
      if (abs(currentYaw) >= mazePivotAngle) {
        stopMotors();
        delay(80);
        resetEncoders(); lockHeading();
        robotState = MAZE_TURN_ADVANCE;
        addLog("MZ pivot done yaw=" + String(currentYaw, 1));
      } else {
        if (mazeTurnDir == 1) {
          setMotors(0, turnSpeed);
          // Right turn
        } else {
          setMotors(turnSpeed, 0);
          // Left turn
        }
      }
      break;
    case MAZE_TURN_ADVANCE:
      if (encoderAvgDistanceMM() >= turnForwardMM) {
        // Update robot direction
        robotDir = mazeNextDir;
        addLog("MZ advance done, dir=" + String(robotDir));
        // Back up to center on node if configured
        if (mazeTurnCenterMM > 0) {
          resetEncoders();
          robotState = MAZE_TURN_CENTER;
        } else {
          // After reverse the robot is at cell center, needs full cell distance
          mazeMoveDist = mazeFromReverse ? nodeDistMM : afterTurnDist;
          mazeFromReverse = false;
          if (mazeMoveDist < 10) mazeMoveDist = 10;
          resetEncoders(); lockHeading();
          robotState = MAZE_FORWARD;
          addLog("MZ fwd=" + String(mazeMoveDist));
        }
      } else {
        driveFused(moveSpeed);
      }
      break;

    case MAZE_TURN_CENTER:
      if (encoderAvgDistanceMM() >= mazeTurnCenterMM) {
        stopMotors();
        delay(50);
        // After reverse the robot is at cell center, needs full cell distance
        mazeMoveDist = mazeFromReverse ? nodeDistMM : afterTurnDist;
        mazeFromReverse = false;
        if (mazeMoveDist < 10) mazeMoveDist = 10;
        resetEncoders(); lockHeading();
        robotState = MAZE_FORWARD;
        addLog("MZ centered, fwd=" + String(mazeMoveDist));
      } else {
        setMotors(-moveSpeed, -moveSpeed);
      }
      break;

    case MAZE_FORWARD:
      if (encoderAvgDistanceMM() >= mazeMoveDist) {
        // Arrived at distance target
        stopMotors();
        // Update position to next cell
        robotRow += DR[robotDir];
        robotCol += DC[robotDir];
        robotRow = constrain(robotRow, 0, 7);
        robotCol = constrain(robotCol, 0, 7);
        addLog("MZ arrived (" + String(robotRow) + "," + String(robotCol) + ")");
        // Go to scan
        mazePauseStart = millis();
        robotState = MAZE_SCAN;
      }
      else if (distFront > 0 && distFront < frontStopDist) {
        // Front obstacle detected mid-move! Emergency handling
        stopMotors();
        addLog("MZ front wall mid-move! d=" + String(distFront));
        // Mark front wall at current position
        setWallBilateral(robotRow, robotCol, robotDir);
        // Stay at current cell, re-decide
        mazePauseStart = millis();
        robotState = MAZE_SCAN;
      }
      else {
        driveFused(moveSpeed);
      }
      break;

    case MAZE_REVERSE:
      // Drive backward one cell distance
      if (encoderAvgDistanceMM() >= mazeMoveDist) {
        stopMotors();
        // Update position: move in the OPPOSITE of robotDir (going backward)
        int backDir = (robotDir + 2) % 4;
        robotRow += DR[backDir];
        robotCol += DC[backDir];
        robotRow = constrain(robotRow, 0, 7);
        robotCol = constrain(robotCol, 0, 7);
        // DO NOT flip robotDir — we're still facing the same way, just backing up
        addLog("MZ reversed to (" + String(robotRow) + "," + String(robotCol) + ")");
        addToPath();

        // Scan walls at this cell and re-run floodfill
        scanMazeWalls();
        floodFill();
        // Check if a non-180 route exists from here
        int checkBest = findBestDirection();
        if (checkBest == -1 || mazeFlood[robotRow][robotCol] == 255) {
          robotState = STOPPED;
          stopMotors();
          mazeActive = false;
          addLog("MZ: No path to target!");
        } else {
          int checkTurn = (checkBest - robotDir + 4) % 4;
          if (checkTurn == 2) {
            // Still needs 180 — keep reversing
            mazeMoveDist = nodeDistMM;
            resetEncoders(); lockHeading();
            addLog("MZ keep reversing");
            // stay in MAZE_REVERSE
          } else {
            // Found a non-180 route — pause, scan properly, then decide
            mazeFromReverse = true;
            mazePauseStart = millis();
            robotState = MAZE_DECIDE;
            addLog("MZ reverse done, can go dir=" + String(checkBest));
          }
        }
      } else {
        driveBackwardFused(moveSpeed);
      }
      break;

    case MAZE_DONE:
      setMotors(0, 0);
      // Stay here until user stops or resets
      break;
  }
}
